SELECT names
FROM city
WHERE rating > 3
AND name != 'Berlin'
AND name != 'Madrid'
AND name LIKE '_ublin'
AND population BETWEEN 400 AND 430
AND rating IS NOT NULL
AND country_id IN (1,2,3,4)
ORDER BY rating DESC;
